Markeaze js tracker

Wiki

Options

Format: mkz([function], [mixData]);

  • Set tracker endpoint
    mkz('endpoint', 'tracker.markeaze.com');
  • Set app key
    mkz('appKey', 'yyyyyyyyyyyyyyyyyyyyyyy');
  • Debug mode
    mkz('debug', true);
  • Callback on ready
    mkz(function() {
  console.log('On ready');
});
  • Set visitor info
    mkz('visitorInfoSet', {
  first_name: 'Kong',
  last_name: 'Qiu'
});
  • Subscribe to mkz-js callbacks
    Avaliable callbacks: url.change, track.before, track.after
    mkz('subscribe', 'url.change', function(url) {
  mkz('trackPageView');
  console.log('Change url', url);
});

Events

Format: mkz('track' + [eventName], [properties], [callbackFunction]);

  • Page view
    mkz('trackPageView');
  • Category view
    mkz('trackPageView', {
  category: {
    id: 1
  }
});
  • Offer view
    mkz('trackPageView', {
  offer: {
    id: 'bb1',
    price: 100.0,
    name: 'Product',
    picture_url: 'http://site.com'
  }
});
  • Cart update
    mkz('trackCartUpdate', {
  items: [
    {
      offer_id: 'bb1',
      qnt: 2.0,
      price: 100.0
    }
  ]
});
  • Cart add item
    mkz('trackCartAddItem', {
  item: {
    offer_id: 'bb1',
    qnt: 2.0,
    price: 100.0
  }
});
  • Cart remove item
    mkz('trackCartRemoveItem', {
  item: {
    offer_id: 'bb1',
    qnt: 1.0
  }
});
  • Order create
    mkz('trackOrderCreate', {
  order_id: '123',
  total: 200.0,
  items: [
    {
      offer_id: 'bb1',
      qnt: 2.0,
      price: 100.0
    }
  ]
});
  • Visitor update
    mkz('trackVisitorUpdate', {
  first_name: 'Vladimir',
  last_name: 'Savelyev'
});
  • Web form show
    mkz('trackWebFormShow', {
  web_form_id: 1
});
  • Web form click
    mkz('trackWebFormClick', {
  web_form_id: 1,
  action_type: 'open_link',
  link_url: 'http://site.com'
});
  • Web form submit
    mkz('trackWebFormSubmit', {
  web_form_id: 1,
  web_form_data: '{"email":"mail@example.net"}'
});
  • Web form close
    mkz('trackWebFormClose', {
  web_form_id: 1
});
